title UserLogsIn

participant "User" as U
participant "Students&Companies" as SC

activate U
activate SC
U->SC:request(loginPage)

alt Validation successful

SC-->U:show(homePage)

else Validation failed

SC-->U:error(wrongCredentials)
SC-->U:show(loginPage)

end
