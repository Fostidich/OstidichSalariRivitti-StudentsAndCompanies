participant "WebServer" as WS
participant "RequestDispatcher" as RD
participant "AuthenticationService" as AS
participant "EmailService" as ES
participant "QueryService" as QS

activate BI
activate WS

activate RD

activate AS

activate QS

deactivate QS

deactivate ES

deactivate AS
WS<--RD:201
deactivate RD

deactivate BI
